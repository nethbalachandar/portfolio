Project Overview

This project explores the use of Graph Neural Networks (GNNs) to predict material properties and support data-driven materials selection. GNNs capture structural relationships between atoms and bonds to model physical behavior in materials science.

Problem Addressed

Material discovery and optimization are bottlenecked by costly experiments and simulations. Classical ML struggles to model crystalline structures effectively.

Solution Approach

GNNs offer a way to directly learn from atomic graphs to predict mechanical, electronic, or thermodynamic properties with high accuracy.

Key Innovation

Leveraging message-passing neural networks to capture complex atomic interactions and structural dependencies in crystalline materials.

def structure_to_graph(structure):
    # Generate atom features
    atom_features = []
    for site in structure:
        features = [site.specie.atomic_mass, site.specie.atomic_radius, ...]
        atom_features.append(features)
    
    # Generate edge features (bonds)
    edges = structure.get_all_neighbors(r=5.0)
    edge_indices = []
    edge_features = []

Model Evaluation Pipeline
# Training and evaluation
from sklearn.metrics import mean_absolute_error, r2_score
import numpy as np

def train_model(model, train_loader, val_loader, epochs=200):
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    criterion = nn.MSELoss()
    
    for epoch in range(epochs):
        model.train()
        total_loss = 0
        
        for batch in train_loader:
            optimizer.zero_grad()
            out = model(batch)
            loss = criterion(out.squeeze(), batch.y)
            loss.backward()
            optimizer.step()
            total_loss += loss.item()
        
        # Validation
        if epoch % 20 == 0:
            val_loss = evaluate_model(model, val_loader)
            print(f'Epoch {epoch}, Train Loss: {total_loss/len(train_loader):.4f}, Val Loss: {val_loss:.4f}')

def evaluate_model(model, test_loader):
    model.eval()
    predictions = []
    targets = []
    
    with torch.no_grad():
        for batch in test_loader:
            out = model(batch)
            predictions.extend(out.squeeze().cpu().numpy())
            targets.extend(batch.y.cpu().numpy())
    
    mae = mean_absolute_error(targets, predictions)
    r2 = r2_score(targets, predictions)
    
    return mae, r2
